add_namespace = generic

#Gives African Nations Tribal Equipment
country_event = {
		id = generic.1
		title = generic.1.t
		desc = generic.1.d

		hidden = yes
		
		trigger = {
				date > 1857.6.1
				has_idea = AF_T_unorganized_tribal_nation
				NOT = { has_tech = tribaltech }
		}
		
		option = {
				name = generic.1.a
				hidden_effect = {
					set_technology = { tribaltech = 1 }
					set_technology = { infantry_weapons = 0 }
				}
		}
}
