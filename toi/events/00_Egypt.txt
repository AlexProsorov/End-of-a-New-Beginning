	is_triggered_only = yes

		}
	}
	
	option = {#we can do nothing
		name = afr_EGY_vic.1.b
		ai_chance = { factor = 25 }
		add_political_power = -15

			limit = {

			}
			FROM = {
				add_autonomy_ratio = {
					value = 100
					localization = OTO_allows_EGY_army_expansion
				}
			}
		}
		else = {
			FROM = {
				add_political_power = 20
			}
		}
	}
}

country_event = {
	id = afr_EGY_vic.6
	title = afr_EGY_vic.6.t
	desc = afr_EGY_vic.6.d
	#picture = GFX_report_event_generic_riot
	
	fire_only_once = yes
	
	trigger = {
		date > 1862.8.1
		has_country_leader = {
			name = "Muhammad Sa'id Pasha"
			ruling_only = yes
		}
		tag = EGY
	}
	immediate = {
		hidden_effect = {
			retire_country_leader = yes
		}
	}
	mean_time_to_happen={
		days = 150
	}
	
	option = { #Praise Ismail
		name = afr_EGY_vic.6.a
		trigger= {
			OR = {
				has_idea = weak_heir
				has_idea = has_heir
				has_idea = dethroned_heir
			}
		}
		create_country_leader = {
			name = "Isma'il Pasha"
			desc = "POLITICS_GEORGE_V_DESC"
			picture = "Portrait_EGY_Ismail_Pasha.dds"
			expire = "1879.1.1"
			ideology = despotism
			traits = {
				industrial_promoter
			}
		}
		if ={
			limit = {
				has_idea = weak_heir
			}
			add_stability = -0.05
		}
		
	}
	option = { #We have no heir
		name = afr_EGY_vic.6.b
		trigger= {
			OR = {
				has_idea = no_heir
				has_idea = nonfamily_heir
			}
		}
		create_country_leader = {
			name = "Egyptian Regency council"
			desc = "POLITICS_GEORGE_V_DESC"
			picture = "Portrait_EGY_regency.dds"
			expire = "2100.1.1"
			ideology = despotism
			traits = {
			}
		}
		OTO = {
			country_event = {id= afr_EGY_vic.7}
		}
		hidden_effect={
			add_ideas={
				hos_regency
			}
		}
		
	}
	
}
