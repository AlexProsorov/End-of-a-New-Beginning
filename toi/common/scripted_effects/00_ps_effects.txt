increase_ps_by_one = {
	custom_effect_tooltip = PS_tooltip_increase_by_one
	hidden_effect = {
		set_temp_variable = { ROOT.PS_temp1 = ROOT.pol_stability }
		add_to_temp_variable = { ROOT.PS_temp1 = 1 }
		if = {
			limit = {
				check_variable = { temp1 < 4 }
			}
			add_to_variable = {
				var = ROOT.pol_stability
				value = 1
			}
		}
		else = {
			set_variable = {
				var = ROOT.pol_stability
				value = 3
			}
		}
	}
}
