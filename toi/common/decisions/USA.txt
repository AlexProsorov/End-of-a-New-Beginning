					AND = {
						FRA = { NOT = { has_government = social_egalitarianism } }
						FRA = { NOT = { has_government = collectivist_socialism } }
						FRA = { NOT = { has_government = vanguard_communism } }
						OR = {
							ENG = { has_government = social_egalitarianism }
							ENG = { has_government = collectivist_socialism }
							ENG = { has_government = vanguard_communism }
						}
						date > 1875.1.1
					}
				}
				AND = {
					OR = {
						ENG = { has_government = social_egalitarianism }
						ENG = { has_government = collectivist_socialism }
						ENG = { has_government = vanguard_communism }
						}
					OR = {
						FRA = { has_government = social_egalitarianism }
						FRA = { has_government = collectivist_socialism }
						FRA = { has_government = vanguard_communism }
					}
					date > 1870.1.1
				}
				date > 1880.1.1
			}
		}
	}

		}

		ai_will_do = {
			base = 1
			modifier = {
        	    factor = 0
            	is_historical_focus_on = yes
           	}
		}
